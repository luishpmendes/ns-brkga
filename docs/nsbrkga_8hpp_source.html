965  const std::vector<double> &a2,
966  const std::vector<Sense> &senses) {

973  at_least_as_good = false;

975  // a1 is better than a2 in at least one objective,
976  better = true;
977  }
978  }

981  // and is better than a2 in at least one objective,
982  // thus a1 dominates a2
983  return at_least_as_good && better;

987  static std::vector<std::vector<std::pair<std::vector<double>, T>>>
988  nonDominatedSort(
989  const std::vector<std::pair<std::vector<double>, T>> &fitness,
990  const std::vector<Sense> &senses) {
991 
992  std::vector<std::vector<std::pair<std::vector<double>, T>>> result;
993 
994  if (fitness.empty() || senses.empty()) {
995  return result;
996  }
997 
998  result.reserve(fitness.size());
999 
1000  auto comp =
1001  [senses](const std::pair<std::vector<double>, T> &a,
1002  const std::pair<std::vector<double>, T> &b) -> bool {
1003  for (std::size_t i = 0; i < a.first.size(); i++) {
1004  if (Population::betterThan(a.first[i], b.first[i], senses[i])) {
1005  return true;
1006  }
1007  if (Population::betterThan(b.first[i], a.first[i], senses[i])) {
1008  return false;
1009  }
1010  }
1011  // a == b
1012  return false;
1013  };
1014 
1015  std::vector<std::pair<std::vector<double>, T>> sorted_fitness = fitness;
1016  std::sort(sorted_fitness.begin(), sorted_fitness.end(), comp);
1017  result.emplace_back(1, sorted_fitness.front());
1018 
1019  if (senses.size() == 1) {
1020  for (std::size_t i = 1; i < sorted_fitness.size(); i++) {
1021  if (Population::dominates(sorted_fitness[i - 1].first,
1022  sorted_fitness[i].first, senses)) {
1023  result.emplace_back(1, sorted_fitness[i]);
1024  } else {
1025  result.back().push_back(sorted_fitness[i]);
1026  }
1027  }
1028  } else { // senses.size() >= 2
1029  for (std::size_t i = 1; i < sorted_fitness.size(); i++) {
1030  bool isDominated = false;
1031 
1032  // check if the current solution is
1033  // dominated by a solution in the last front
1034  for (std::size_t j = result.back().size(); j > 0; j--) {
1035  if (Population::dominates(result.back()[j - 1].first,
1036  sorted_fitness[i].first,
1037  senses)) {
1038  isDominated = true;
1039  break;
1040  }
1041 
1042  // if there is only 2 objectives, we need to check for
1043  // dominance only with the last element in the last front
1044  if (senses.size() == 2) {
1045  break;
1046  }
1047  }
1048 
1049  // if the current solution is dominated
1050  // by a solution in the last front
1051  if (isDominated) {
1052  // create a new front to put the current solution
1053  result.emplace_back(1, sorted_fitness[i]);
1054  } else {

1056  // dominates the current solution using binary search
1057  std::size_t kMin = 0, kMax = result.size();
1058  while (kMin < kMax) {

1060  isDominated = false;
1061

1063  // dominated by a solution in the k-th front
1064  for (std::size_t j = result[k].size(); j > 0; j--) {
1065  if (Population::dominates(result[k][j - 1].first,
1066  sorted_fitness[i].first,
1067  senses)) {
1068  isDominated = true;
1069  break;
1070  }
1071 
1072  // if there is only 2 objectives, we need to check
1073  // for dominance only with the last solution in the
1074  // k-th front
1075  if (senses.size() == 2) {
1076  break;
1077  }
1078  }
1079 
1080  if (isDominated) {
1081  kMin = k + 1;
1082  } else {
1083  if (k == kMin) {
1084  break;
1085  }
1086 
1087  kMax = k;
1088  }
1089  }
1090 
1091  result[kMin].push_back(sorted_fitness[i]);

1093  }
1094  }

1101  crowdingSort(std::vector<std::pair<std::vector<double>, T>> &fitness) {

1103  aux(fitness.size());

1105

1108  for (unsigned i = 0; i < aux.size(); i++) {
1109  aux[i] = std::make_pair(fitness[i].first[m],

1111  }
1112 
1113  std::sort(aux.begin(), aux.end());
1114 
1115  double fMin = aux.front().first, fMax = aux.back().first;
1116 
1117  distance[aux.front().second.second] =
1118  distance[aux.back().second.second] =
1119  std::numeric_limits<double>::max();
1120 
1121  for (std::size_t i = 1; i + 1 < aux.size(); i++) {
1122  if (fabs(fMax - fMin) <
1123  std::numeric_limits<double>::epsilon()) {
1124  distance[aux[i].second.second] =
1125  std::numeric_limits<double>::max();
1126  } else if (distance[aux[i].second.second] <
1127  std::numeric_limits<double>::max()) {
1128  distance[aux[i].second.second] +=
1129  (aux[i + 1].second.first[m] -
1130  aux[i - 1].second.first[m]) /
1131  (fMax - fMin);
1132  }
1133  }
1134  }

1137  std::transform(aux.begin(), aux.end(), aux.begin(),

1144  aux.begin(), aux.end(),
1145  std::greater<
1146  std::pair<double, std::pair<std::vector<double>, unsigned>>>());

1149  fitness;

3433 template <class Decoder>
3434 void NSBRKGA<Decoder>::mate(const Population &curr, Chromosome &offspring) {
3435  const unsigned P = this->params.total_parents;
3436 
3437  // Precompute bias weights using global population rank of each parent.
3438  this->total_bias_weight = 0.0;
3439  for (const unsigned &i : this->parents_indexes) {
3440  this->total_bias_weight += this->bias_function(i + 1);
3441  }
3442 
3443  switch (this->params.crossover_type) {
3444  //--------------------------------------------------------------
3445  // GEOMETRIC: biased weighted-average crossover.
3446  // For each parent j (0-based), r = parents_indexes[j] + 1 is
3447  // its 1-based rank in the population. The random weight is
3448  // drawn from Uniform(phi(r), phi(r+1)).
3449  //--------------------------------------------------------------
3450  case CrossoverType::GEOMETRIC: {
3451  for (unsigned gene = 0; gene < this->CHROMOSOME_SIZE; ++gene) {
3452  double weighted_sum = 0.0;
3453  double weight_total = 0.0;
3454 
3455  for (unsigned j = 0; j < P; ++j) {
3456  const unsigned r = this->parents_indexes[j] + 1;
3457  const double a = this->bias_function(r);
3458  const double b = this->bias_function(r + 1);
3459  const double lo = std::min(a, b);
3460  const double hi = std::max(a, b);
3461  // w_r ~ Uniform(lo, hi)
3462  const double w = lo + this->rand01() * (hi - lo);
3463 
3464  weighted_sum += w * curr(this->parents_ordered[j].second, gene);
3465  weight_total += w;
3466  }
3467 
3468  offspring[gene] = weighted_sum / weight_total;
3469 
3470  // Performs the polynomial mutation.
3471  this->polynomialMutation(offspring[gene]);
3472  }
3473  break;
3474  }
3475 
3476  //--------------------------------------------------------------
3477  // ROULETTE (default): discrete biased roulette crossover.
3478  //--------------------------------------------------------------
3479  case CrossoverType::ROULETTE:
3480  default: {
3481  for (unsigned gene = 0; gene < this->CHROMOSOME_SIZE; ++gene) {
3482  // Roulette method using global population rank.
3483  unsigned parent = 0;
3484  double cumulative_probability = 0.0;
3485  const double toss = this->rand01();
3486 
3487  do {
3488  // Start parent from 1 because the bias function.
3489  cumulative_probability +=
3490  this->bias_function(this->parents_indexes[parent++] + 1) /
3491  this->total_bias_weight;
3492  } while (cumulative_probability < toss);
3493 
3494  // Decrement parent to the right index, and take the allele.
3495  offspring[gene] =
3496  curr(this->parents_ordered[--parent].second, gene);
3497 
3498  // Performs the polynomial mutation.
3499  this->polynomialMutation(offspring[gene]);
3500  }
3501  break;
3502  }
3503  } // switch crossover_type
3504 }
