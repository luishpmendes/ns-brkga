1053  template <class T>
1054  static std::vector<std::vector<std::pair<std::vector<double>, T>>>
1055  nonDominatedSort(
1056  const std::vector<std::pair<std::vector<double>, T>> &fitness,
1057  const std::vector<Sense> &senses) {
1058 
1059  std::vector<std::vector<std::pair<std::vector<double>, T>>> result;
1060 
1061  if (fitness.empty() || senses.empty()) {
1062  return result;
1063  }
1064 
1065  const std::size_t numObj = senses.size();
1066 
1067  result.reserve(fitness.size());
1068 
1069  // Lexicographic comparator respecting optimisation senses.
1070  // Captured by const-ref to avoid copying the senses vector.
1071  auto comp =
1072  [&senses](const std::pair<std::vector<double>, T> &a,
1073  const std::pair<std::vector<double>, T> &b) -> bool {
1074  for (std::size_t i = 0; i < a.first.size(); i++) {
1075  if (Population::betterThan(a.first[i], b.first[i], senses[i])) {
1076  return true;
1077  }
1078  if (Population::betterThan(b.first[i], a.first[i], senses[i])) {
1079  return false;
1080  }
1081  }
1082 
1083  // a == b
1084  return false;
1085  };
1086 
1087  std::vector<std::pair<std::vector<double>, T>> sorted_fitness = fitness;
1088  std::sort(sorted_fitness.begin(), sorted_fitness.end(), comp);
1089  result.emplace_back(1, sorted_fitness.front());
1090 
1091  if (numObj == 1) {
1092  // Single-objective fast path: consecutive solutions either
1093  // share the current front or start a new one.
1094  for (std::size_t i = 1; i < sorted_fitness.size(); i++) {
1095  if (Population::dominates(sorted_fitness[i - 1].first,
1096  sorted_fitness[i].first, senses)) {
1097  result.emplace_back(1, sorted_fitness[i]);
1098  } else {
1099  result.back().push_back(sorted_fitness[i]);
1100  }
1101  }
1102  } else { // numObj >= 2
1103  // Helper: returns true if any solution in result[frontIdx]
1104  // dominates sorted_fitness[solIdx]. For 2 objectives only
1105  // the last element of the front needs to be checked (the
1106  // pre-sort guarantees it is the only possible dominator).
1107  auto isDominatedByFront = [&result, &sorted_fitness, &senses,
1108  numObj](std::size_t frontIdx,
1109  std::size_t solIdx) -> bool {
1110  const auto &front = result[frontIdx];
1111 
1112  for (std::size_t j = front.size(); j > 0; j--) {
1113  if (Population::dominates(front[j - 1].first,
1114  sorted_fitness[solIdx].first,
1115  senses)) {
1116  return true;
1117  }
1118 
1119  // For 2 objectives, only the last element can dominate.
1120  if (numObj == 2) {
1121  break;
1122  }
1123  }
1124 
1125  return false;
1126  };
1127 
1128  for (std::size_t i = 1; i < sorted_fitness.size(); i++) {
1129  // Quick check: if the last front dominates this solution,
1130  // it must go into a brand-new front.
1131  if (isDominatedByFront(result.size() - 1, i)) {
1132  result.emplace_back(1, std::move(sorted_fitness[i]));
1133  } else {
1134  // Binary search for the first front that does not
1135  // dominate the current solution.
1136  std::size_t kMin = 0, kMax = result.size();
1137 
1138  while (kMin < kMax) {
1139  std::size_t k = (kMin + kMax) >> 1;
1140 
1141  if (isDominatedByFront(k, i)) {
1142  kMin = k + 1;
1143  } else {
1144  if (k == kMin) {
1145  break;
1146  }
1147 
1148  kMax = k;
1149  }
1150  }
1151 
1152  result[kMin].push_back(std::move(sorted_fitness[i]));
1153  }
1154  }
1155  }
1156

3495 template <class Decoder>
3496 void NSBRKGA<Decoder>::mate(const Population &curr, Chromosome &offspring) {
3497  const unsigned P = this->params.total_parents;
3498 
3499  // Precompute bias weights using global population rank of each parent.
3500  this->total_bias_weight = 0.0;
3501  for (const unsigned &i : this->parents_indexes) {
3502  this->total_bias_weight += this->bias_function(i + 1);
3503  }
3504 
3505  switch (this->params.crossover_type) {
3506  //--------------------------------------------------------------
3507  // GEOMETRIC: biased weighted-average crossover.
3508  // For each parent j (0-based), r = parents_indexes[j] + 1 is
3509  // its 1-based rank in the population. The random weight is
3510  // drawn from Uniform(phi(r), phi(r+1)).
3511  //--------------------------------------------------------------
3512  case CrossoverType::GEOMETRIC: {
3513  for (unsigned gene = 0; gene < this->CHROMOSOME_SIZE; ++gene) {
3514  double weighted_sum = 0.0;
3515  double weight_total = 0.0;
3516 
3517  for (unsigned j = 0; j < P; ++j) {
3518  const unsigned r = this->parents_indexes[j] + 1;
3519  const double a = this->bias_function(r);
3520  const double b = this->bias_function(r + 1);
3521  const double lo = std::min(a, b);
3522  const double hi = std::max(a, b);
3523  // w_r ~ Uniform(lo, hi)
3524  const double w = lo + this->rand01() * (hi - lo);
3525 
3526  weighted_sum += w * curr(this->parents_ordered[j].second, gene);
3527  weight_total += w;
3528  }
3529 
3530  offspring[gene] = weighted_sum / weight_total;
3531 
3532  // Performs the polynomial mutation.
3533  this->polynomialMutation(offspring[gene]);
3534  }
3535  break;
3536  }
3537 
3538  //--------------------------------------------------------------
3539  // ROULETTE (default): discrete biased roulette crossover.
3540  //--------------------------------------------------------------
3541  case CrossoverType::ROULETTE:
3542  default: {
3543  for (unsigned gene = 0; gene < this->CHROMOSOME_SIZE; ++gene) {
3544  // Roulette method using global population rank.
3545  unsigned parent = 0;
3546  double cumulative_probability = 0.0;
3547  const double toss = this->rand01();
3548 
3549  do {
3550  // Start parent from 1 because the bias function.
3551  cumulative_probability +=
3552  this->bias_function(this->parents_indexes[parent++] + 1) /
3553  this->total_bias_weight;
3554  } while (cumulative_probability < toss);
3555 
3556  // Decrement parent to the right index, and take the allele.
3557  offspring[gene] =
3558  curr(this->parents_ordered[--parent].second, gene);
3559 
3560  // Performs the polynomial mutation.
3561  this->polynomialMutation(offspring[gene]);
3562  }
3563  break;
3564  }
3565  } // switch crossover_type
3566 }
