1053  template <class T>
1054  static std::vector<std::vector<std::pair<std::vector<double>, T>>>
1055  nonDominatedSort(
1056  const std::vector<std::pair<std::vector<double>, T>> &fitness,
1057  const std::vector<Sense> &senses) {
1058 
1059  std::vector<std::vector<std::pair<std::vector<double>, T>>> result;
1060 
1061  if (fitness.empty() || senses.empty()) {
1062  return result;
1063  }
1064 
1065  const std::size_t numObj = senses.size();
1066 
1067  result.reserve(fitness.size());
1068 
1069  // Lexicographic comparator respecting optimisation senses.
1070  // Captured by const-ref to avoid copying the senses vector.
1071  auto comp =
1072  [&senses](const std::pair<std::vector<double>, T> &a,
1073  const std::pair<std::vector<double>, T> &b) -> bool {
1074  for (std::size_t i = 0; i < a.first.size(); i++) {
1075  if (Population::betterThan(a.first[i], b.first[i], senses[i])) {
1076  return true;
1077  }
1078  if (Population::betterThan(b.first[i], a.first[i], senses[i])) {
1079  return false;
1080  }
1081  }
1082 
1083  // a == b
1084  return false;
1085  };
1086 
1087  std::vector<std::pair<std::vector<double>, T>> sorted_fitness = fitness;
1088  std::sort(sorted_fitness.begin(), sorted_fitness.end(), comp);
1089  result.emplace_back(1, sorted_fitness.front());
1090 
1091  if (numObj == 1) {
1092  // Single-objective fast path: consecutive solutions either
1093  // share the current front or start a new one.
1094  for (std::size_t i = 1; i < sorted_fitness.size(); i++) {
1095  if (Population::dominates(sorted_fitness[i - 1].first,
1096  sorted_fitness[i].first, senses)) {
1097  result.emplace_back(1, sorted_fitness[i]);
1098  } else {
1099  result.back().push_back(sorted_fitness[i]);
1100  }
1101  }
1102  } else { // numObj >= 2
1103  // Helper: returns true if any solution in result[frontIdx]
1104  // dominates sorted_fitness[solIdx]. For 2 objectives only
1105  // the last element of the front needs to be checked (the
1106  // pre-sort guarantees it is the only possible dominator).
1107  auto isDominatedByFront = [&result, &sorted_fitness, &senses,
1108  numObj](std::size_t frontIdx,
1109  std::size_t solIdx) -> bool {
1110  const auto &front = result[frontIdx];
1111 
1112  for (std::size_t j = front.size(); j > 0; j--) {
1113  if (Population::dominates(front[j - 1].first,
1114  sorted_fitness[solIdx].first,
1115  senses)) {
1116  return true;
1117  }
1118 
1119  // For 2 objectives, only the last element can dominate.
1120  if (numObj == 2) {
1121  break;
1122  }
1123  }
1124 
1125  return false;
1126  };
1127 
1128  for (std::size_t i = 1; i < sorted_fitness.size(); i++) {
1129  // Quick check: if the last front dominates this solution,
1130  // it must go into a brand-new front.
1131  if (isDominatedByFront(result.size() - 1, i)) {
1132  result.emplace_back(1, std::move(sorted_fitness[i]));
1133  } else {
1134  // Binary search for the first front that does not
1135  // dominate the current solution.
1136  std::size_t kMin = 0, kMax = result.size();
1137 
1138  while (kMin < kMax) {
1139  std::size_t k = (kMin + kMax) >> 1;
1140 
1141  if (isDominatedByFront(k, i)) {
1142  kMin = k + 1;
1143  } else {
1144  if (k == kMin) {
1145  break;
1146  }
1147 
1148  kMax = k;
1149  }
1150  }
1151 
1152  result[kMin].push_back(std::move(sorted_fitness[i]));
1153  }
1154  }
1155  }
1156 
1157  return result;
1158  }

1189  template <class T>
1190  static void
1191  crowdingSort(std::vector<std::pair<std::vector<double>, T>> &fitness) {
1192  const std::size_t n = fitness.size();
1193 
1194  if (n <= 1) {
1195  return; // Nothing to sort.
1196  }
1197 
1198  const std::size_t numObj = fitness.front().first.size();
1199 
1200  // distance[i] accumulates the crowding distance for fitness[i].
1201  std::vector<double> distance(n, 0.0);
1202 
1203  // order[i] is an index into fitness[]; sorted per-objective below.
1204  std::vector<unsigned> order(n);
1205 
1206  for (unsigned i = 0; i < n; i++) {
1207  order[i] = i;
1208  }
1209 
1210  // --- Step 2: accumulate crowding distance for each objective --------
1211  for (std::size_t m = 0; m < numObj; m++) {
1212  // Sort indices by the m-th objective value.
1213  std::sort(order.begin(), order.end(),
1214  [&fitness, m](unsigned a, unsigned b) {
1215  return fitness[a].first[m] < fitness[b].first[m];
1216  });
1217 
1218  const double fMin = fitness[order.front()].first[m];
1219  const double fMax = fitness[order.back()].first[m];
1220 
1221  // Boundary solutions always receive infinite distance.
1222  distance[order.front()] = std::numeric_limits<double>::max();
1223  distance[order.back()] = std::numeric_limits<double>::max();
1224 
1225  // Interior solutions.
1226  for (std::size_t i = 1; i + 1 < n; i++) {
1227  const unsigned idx = order[i];
1228 
1229  if (fabs(fMax - fMin) <
1230  std::numeric_limits<double>::epsilon()) {
1231  // Degenerate range: all solutions are equally spread.
1232  distance[idx] = std::numeric_limits<double>::max();
1233  } else if (distance[idx] < std::numeric_limits<double>::max()) {
1234  // Accumulate normalised crowding span from neighbours.
1235  distance[idx] += (fitness[order[i + 1]].first[m] -
1236  fitness[order[i - 1]].first[m]) /
1237  (fMax - fMin);
1238  }
1239  }
1240  }
1241 
1242  // --- Step 3: sort by descending (distance, index) -------------------
1243  // After the last objective loop, `order` holds the sorted-by-last-
1244  // objective permutation. Building aux from `order` preserves the
1245  // same secondary-sort key as the original implementation.
1246  std::vector<std::pair<double, unsigned>> aux(n);
1247 
1248  for (std::size_t i = 0; i < n; i++) {
1249  aux[i] = {distance[order[i]], order[i]};
1250  }
1251 
1252  std::sort(aux.begin(), aux.end(),
1253  std::greater<std::pair<double, unsigned>>());
1254 
1255  // --- Step 4: permute fitness in-place -------------------------------
1256  // Build the target permutation: perm[i] = source index for slot i.
1257  std::vector<unsigned> perm(n);
1258 
1259  for (std::size_t i = 0; i < n; i++) {
1260  perm[i] = aux[i].second;
1261  }
1262 
1263  // Follow cycles to move elements with O(n) swaps and no full copy.
1264  std::vector<bool> placed(n, false);
1265 
1266  for (std::size_t i = 0; i < n; i++) {
1267  if (placed[i] || perm[i] == i) {
1268  continue;
1269  }
1270 
1271  std::size_t j = i;
1272 
1273  while (!placed[j]) {
1274  placed[j] = true;
1275 
1276  if (perm[j] != i) {
1277  std::swap(fitness[j], fitness[perm[j]]);
1278  j = perm[j];
1279  } else {
1280  // Close the cycle: element at perm[j] (== i) is already
1281  // in its final position after the preceding swaps.
1282  j = perm[j];
1283  }
1284  }
1285  }
1286  }

1330  template <class T>

1332  sortFitness(std::vector<std::pair<std::vector<double>, T>> &fitness,
1333  const std::vector<Sense> &senses) {

1335  return std::make_pair(0, 0);
1336  }
1337

1339  if (senses.size() == 1) {
1340  std::sort(fitness.begin(), fitness.end(),
1341  [&senses](const std::pair<std::vector<double>, T> &a,
1342  const std::pair<std::vector<double>, T> &b) {
1343  return Population::betterThan(
1344  a.first.front(), b.first.front(), senses.front());

3615 void NSBRKGA<Decoder>::mate(const Population &curr, Chromosome &offspring) {

3619  this->total_bias_weight = 0.0;
3620  for (const unsigned &i : this->parents_indexes) {
3621  this->total_bias_weight += this->bias_function(i + 1);
3622  }
3623

3626  // GEOMETRIC: biased weighted-average crossover.
3627  // For each parent j (0-based), r = parents_indexes[j] + 1 is
3628  // its 1-based rank in the population. The random weight is
3629  // drawn from Uniform(phi(r), phi(r+1)).
3630  //--------------------------------------------------------------
3631  case CrossoverType::GEOMETRIC: {

3633  double weighted_sum = 0.0;

3635 
3636  for (unsigned j = 0; j < P; ++j) {
3637  const unsigned r = this->parents_indexes[j] + 1;
3638  const double a = this->bias_function(r);
3639  const double b = this->bias_function(r + 1);
3640  const double lo = std::min(a, b);
3641  const double hi = std::max(a, b);
3642  // w_r ~ Uniform(lo, hi)
3643  const double w = lo + this->rand01() * (hi - lo);
3644 
3645  weighted_sum += w * curr(this->parents_ordered[j].second, gene);
3646  weight_total += w;
3647  }
3648 
3649  offspring[gene] = weighted_sum / weight_total;

3651  // Performs the polynomial mutation.
3652  this->polynomialMutation(offspring[gene]);
3653  }
3654  break;
3655  }

3657  //--------------------------------------------------------------
3658  // ROULETTE (default): discrete biased roulette crossover.
3659  //--------------------------------------------------------------
3660  case CrossoverType::ROULETTE:
3661  default: {
3662  for (unsigned gene = 0; gene < this->CHROMOSOME_SIZE; ++gene) {
3663  // Roulette method using global population rank.

3665  double cumulative_probability = 0.0;

3670  cumulative_probability +=
3671  this->bias_function(this->parents_indexes[parent++] + 1) /

3676  offspring[gene] =
3677  curr(this->parents_ordered[--parent].second, gene);
3678

3680  this->polynomialMutation(offspring[gene]);
3681  }
3682  break;
3683  }
3684  } // switch crossover_type
3685 }
